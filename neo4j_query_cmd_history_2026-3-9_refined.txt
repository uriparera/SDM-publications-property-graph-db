RETURN row LIMIT 1;

CREATE CONSTRAINT FOR (p:Paper) REQUIRE p.id IS UNIQUE;
CREATE CONSTRAINT FOR (a:Author) REQUIRE a.name IS UNIQUE;
CREATE CONSTRAINT FOR (k:Keyword) REQUIRE k.word IS UNIQUE;

// Load Inproceedings
LOAD CSV WITH HEADERS FROM 'file:///clean_inproceedings.csv' AS row FIELDTERMINATOR ';'
MERGE (p:Paper {id: row.`inproceedings:ID`})
SET p.title = row.`title:string`, 
    p.year = toInteger(toFloat(row.`year:int`)), 
    p.type = 'Inproceedings';

// Load Articles
LOAD CSV WITH HEADERS FROM 'file:///clean_articles.csv' AS row FIELDTERMINATOR ';'
MERGE (p:Paper {id: row.`article:ID`})
SET p.title = row.`title:string`, 
    p.year = toInteger(toFloat(row.`year:int`)), 
    p.type = 'Article';

// Load Authors Safely

// Load Keywords
LOAD CSV WITH HEADERS FROM 'file:///nodes_keyword.csv' AS row FIELDTERMINATOR ';'

LOAD CSV WITH HEADERS FROM 'file:///rels_written_by_TOTAL.csv' AS row FIELDTERMINATOR ';'

// 1. Create the new correctly directed relationships
MATCH (a:Author)-[old:WRITTEN_BY]->(p:Paper)
MERGE (p)-[:WRITTEN_BY {corresponding: old.corresponding}]->(a)
DELETE old;

LOAD CSV WITH HEADERS FROM 'file:///nodes_edition.csv' AS row FIELDTERMINATOR ';'
MERGE (:Edition {id: row.`edition_id:ID`, city: row.`city:string`, year: toInteger(row.`year:int`)});

// 1. Create Journal and Volume nodes
LOAD CSV WITH HEADERS FROM 'file:///nodes_journal.csv' AS row FIELDTERMINATOR ';'
MERGE (:Journal {name: row.`name:ID`});

// 2. Create Conference and Edition nodes
LOAD CSV WITH HEADERS FROM 'file:///nodes_conference.csv' AS row FIELDTERMINATOR ';'
MERGE (:Conference {name: row.`name:ID`});

LOAD CSV WITH HEADERS FROM 'file:///nodes_volume.csv' AS row FIELDTERMINATOR ';'
MERGE (:Volume {id: row.`volume_id:ID`, number: toInteger(row.`number:int`), year: toInteger(row.`year:int`)});

// Or updating to add year
LOAD CSV WITH HEADERS FROM 'file:///nodes_volume.csv' AS row FIELDTERMINATOR ';'

    v.year = toInteger(row.`year:int`);

// Link Inproceedings to Editions
LOAD CSV WITH HEADERS FROM 'file:///rels_presented_at.csv' AS row FIELDTERMINATOR ';'
MATCH (p:Paper {id: row.`:START_ID`}), (e:Edition {id: row.`:END_ID`})
MERGE (p)-[:PRESENTED_AT]->(e);

// Link Articles to Volumes
LOAD CSV WITH HEADERS FROM 'file:///rels_published_in.csv' AS row FIELDTERMINATOR ';'
MATCH (p:Paper {id: row.`:START_ID`}), (v:Volume {id: row.`:END_ID`})
MERGE (p)-[:PUBLISHED_IN]->(v);

// Edition -> BELONGS_TO -> Conference
LOAD CSV WITH HEADERS FROM 'file:///rels_edition_belongs_to.csv' AS row FIELDTERMINATOR ';'
MATCH (e:Edition {id: row.`:START_ID`}), (c:Conference {name: row.`:END_ID`})
MERGE (e)-[:BELONGS_TO]->(c);

// Volume -> BELONGS_TO -> Journal
LOAD CSV WITH HEADERS FROM 'file:///rels_volume_belongs_to.csv' AS row FIELDTERMINATOR ';'
MATCH (v:Volume {id: row.`:START_ID`}), (j:Journal {name: row.`:END_ID`})
MERGE (v)-[:BELONGS_TO]->(j);

MATCH (j:Journal)<-[:BELONGS_TO]-(v:Volume)<-[:PUBLISHED_IN]-(p:Paper)-[:WRITTEN_BY]->(a:Author)
RETURN j, v, p, a LIMIT 5;

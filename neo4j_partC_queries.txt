// Stage 1. The first thing to do is to find/define the research communities. A community is defined by a set of keywords. Assume that the database community is defined through the following keywords: data management, indexing, data modeling, big data, data processing, data storage and data querying.

MERGE (c:Community {name: "Database"}) 
WITH c
UNWIND ["data management", "indexing", "data modeling", "big data", "data processing", "data storage", "data querying"] AS db_keywords 
MATCH (k:Keyword {word: db_keywords})
MERGE (c)-[:DEFINED_BY]->(k) 
RETURN c.name AS Community, collect(k.word) AS LinkedKeywords // optional return sequence to check

// ensures no duplicates
// treat list of keywords
// find actual keywords and are linked to community

// Stage 2. Next, we need to find the conferences, workshops and journals related to the database community (i.e., those that are specific to the field of databases). Assume that if 90% of the papers published in a conference/workshop/journal contain one of the keywords of the database community we consider that conference/workshop/journal as related to that community.

MATCH (com:Community {name: "Database"})-[:DEFINED_BY]->(kw:Keyword) // finds corresponding keywords
WITH com, collect(kw) AS communityKeywords // groups those into a list for comparison
MATCH (p:Paper)-[:PRESENTED_AT|PUBLISHED_IN]->()-[:BELONGS_TO]->(v) 
WHERE v:Conference OR v:Journal
WITH com, v, communityKeywords, collect(DISTINCT p) AS allPapers // papers ever participating in that venue
WITH com, v, size(allPapers) AS total, 
     [paper IN allPapers WHERE EXISTS { (paper)-[:COVERS]->(k) WHERE k IN communityKeywords }] AS communityPapers // papers that actually cover at least one topic related to community
WITH com, v, total, size(communityPapers) AS countComm
WHERE total > 0 AND (toFloat(countComm) / total) >= 0.9 // apply threshold 
MERGE (v)-[:RELATED_TO]->(com) // create relationship 
RETURN v.name AS Venue, total AS TotalPapers, countComm AS CommunityPapers; // optional return sequence to check

// Stage 3. Next, we want to identify the top papers of these conferences/workshops/journals. We need to find the 100 papers with the highest number of citations from papers of the database community). As a result we will obtain the top-100 papers of the database community.

// a Gemini no li ha molat aquesta query que havia fet jo :(
MATCH (v)-[:RELATED_TO]->(c:Community {name: "Database"})
WHERE v:Conference OR v:Journal
WITH v // retrieve venues related to such community
MATCH (p2:Paper)-[:PRESENTED_AT|PUBLISHED_IN]->()-[:BELONGS_TO]->(v) // retrieve papers from such venues
MATCH (p1:Paper)-[:CITED_BY]->(p2:Paper)
WITH v, p1, COUNT(p2) as citations
ORDER BY citations DESC
WITH v, collect({p1, citations: citations})[0..100] as top100Papers
UNWIND top100Papers as top100Paper
SET top100Paper.isTop100 = True, top100Paper:TopPaper
RETURN p1.title AS Title, citations AS Citations;

MATCH (v)-[:RELATED_TO]->(c:Community {name: "Database"})
MATCH (p1:Paper)-[:PRESENTED_AT|PUBLISHED_IN]->()-[:BELONGS_TO]->(v) // retrieve papers from such venues
MATCH (p2:Paper)-[:CITED_BY]->(p1)
WHERE EXISTS {MATCH (p2)-[:PRESENTED_AT|PUBLISHED_IN]->()-[:BELONGS_TO]->()-[:RELATED_TO]->(c)}
// if we wrote belongs_to v, then only the citations from the same venue would count
WITH p2, count(p1) AS citations 
ORDER BY citations DESC
LIMIT 100 // extract 100 first
SET p2:TopPaper, p2.isTop100 = true // add new label to node and property, maybe one would be enough
RETURN p2.title AS Title, citations AS Citations;

// To remove any wrong query consequences 
MATCH (p)
WHERE p:TopPaper OR p.isTop100 IS NOT NULL
REMOVE p:TopPaper
SET p.isTop100 = null
RETURN count(p) AS CleanedNodes;

// Stage 4. Finally, an author of any of these top-100 papers is automatically considered a potential good match to review database papers, and we want to include this information in the graph. In addition, we want to identify and store gurus, i.e., reputable authors that would be able to review for top events or journals. Gurus are those authors of, at least, two papers among the top-100 identified.

MATCH (p:TopPaper)-[:WRITTEN_BY]->(a:Author)
WITH a, count(p) AS writtenPapers
SET a:PotentialReviewer
WITH a, writtenPapers
WHERE writtenPapers >= 2 
SET a:Guru
RETURN a.name AS Author, writtenPapers;
